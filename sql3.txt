example 4

SELECT distinct 
    emp.first_name || ' ' || emp.last_name AS full_name,
    emp.department_id, 
    jobs.job_title,
    MAX(emp.salary) AS max_salary,
    MIN(emp.salary) AS min_salary,
    dep.department_name
FROM 
    hr.employees emp
JOIN 
    hr.jobs jobs ON emp.job_id = jobs.job_id
JOIN 
    hr.departments dep ON emp.department_id = dep.department_id
WHERE 
    jobs.job_id LIKE 'S%'
GROUP BY 
    emp.first_name, 
    emp.last_name, 
    emp.department_id, 
    jobs.job_title,
    dep.department_name;
